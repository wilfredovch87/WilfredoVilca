\textbf{}\\
\textbf{}\\

\textbf{}\\
\textbf{}\\
SELECT\\
custgroup,\\
country,\\
numberofcustomers\\
FROM Sales.PivotCustGroups\\
UNPIVOT (numberofcustomers FOR custgroup IN (A, B, C)) AS p;
